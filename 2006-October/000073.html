Everyone should start working with the UseOA-ROSE trunk now.  Top on  
Priya's list will be to get the regression tests working that are  
currently broken.  See the log entry for what needs fixed.

The merge includes some upgrades to the makefiles.  Andy is working  
on getting the docs up on the wiki, but here is the script I used on  
okapi to get this guy to compile.  Notice especially that on okapi  
everything needs to be compiled with gcc 3.2.  Also, notice that I  
just installed a new version of ROSE.

Thanks,
Michelle

setenv GCCDIR /home/software/gcc-3.2
setenv GCCVER 3.2

setenv CC $GCCDIR/bin/gcc
setenv CXX $GCCDIR/bin/g++
#setenv CXXFLAGS '-fpermissive'
unalias gcc
unalias g++

# Do not modify next two lines
set path=($GCCDIR/bin $GCCDIR/lib $path)
if ($?LD_LIBRARY_PATH) then
   setenv LD_LIBRARY_PATH $GCCDIR/lib:$LD_LIBRARY_PATH
else
   setenv LD_LIBRARY_PATH $GCCDIR/lib
endif

# Specifies what debugging & profile flags are sent to the compiler
# and linker
#PROFILE_OPTIONS = -O0 -g -pg
setenv PROFILE_OPTIONS '-O0 -g'
#PROFILE_OPTIONS =

# Top-level UseOA source directory
#setenv topsrcdir /home/mstrout/oadev/rose/gcc-3.2/UseOA-ROSE
setenv USEOA_ROSE_USEOAROSEDIR /home/mstrout/oadev/rose/gcc-3.2/UseOA- 
ROSE

# Root of ROSE installation
#setenv ROSE_INST_ROOT /home/mstrout/ROSE/rose-0.8.7a/compile-gcc-3.2
setenv USEOA_ROSE_ROSEDIR /home/mstrout/ROSE/rose-0.8.9a/compile-gcc-3.2

# OpenAnalysis variables
#setenv OPENANALYSIS_DIR /home/mstrout/oadev/rose/gcc-3.2/OpenAnalysis
setenv USEOA_ROSE_OADIR /home/mstrout/oadev/rose/gcc-3.2/OpenAnalysis

setenv USEOA_ROSE_ARCH x86-Linux
#setenv OPENANALYSIS_ARCH i686-Linux
#OPENANALYSIS_ARCH=opteron-Linux

# Compiler/linker
setenv CXX g++
setenv CXXLD g++
setenv CXXFLAGS '-g -O0'
setenv USEOA_ROSE_CXXFLAGS '-g -O0'
